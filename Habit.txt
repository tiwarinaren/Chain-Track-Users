        // Check if habit was already tracked today
        function hasTrackedToday(habitId) {
            const today = new Date().toDateString();
            const lastTracked = state.lastTrackedDate[`${state.currentUser}-${habitId}`];
            return lastTracked === today;
        }

        // Calculate streak
        function calculateStreak(chain) {
            let streak = 0;
            for (let i = chain.length - 1; i >= 0; i--) {
                if (chain[i]) {
                    streak++;
                } else {
                    break;
                }
            }
            return streak;
        }

        // Calculate completion rate
        function calculateCompletionRate(chain) {
            const completed = chain.filter(day => day).length;
            return (completed / (completed + chain.filter(day => !day).length)) * 100;
        }

        // Update current streak display
        function updateCurrentStreak() {
            const userHabits = state.habits[state.currentUser];
            if (userHabits.length === 0) {
                streakDays.textContent = '0';
                streakIcon.classList.add('hidden');
                return;
            }
            
            const streaks = userHabits.map(habit => calculateStreak(habit.chain));
            const maxStreak = Math.max(...streaks);
            streakDays.textContent = maxStreak;
            
            if (maxStreak > 3) {
                streakIcon.classList.remove('hidden');
            } else {
                streakIcon.classList.add('hidden');
            }
        }

                                        ${latestStreak2} day${latestStreak2 !== 1 ? 's' : ''}
                                        ${latestStreak2 > 7 ? '<i class="fas fa-fire text-yellow-500 ml-1 streak-fire"></i>' : ''}
                                    </p>
                                    <p class="text-sm text-gray-600 dark:text-gray-300">${habit.user2.chain.filter(d => d).length} of ${habit.user2.chain.length} days</p>
                                ` : `
                                    <p class="text-sm text-gray-500 dark:text-gray-400 italic">Habit not added</p>
                                    <p class="text-xs text-gray-400 dark:text-gray-500 mt-1">This user hasn't added this habit yet</p>
                                `}
                            </div>
                        </div>
                    </div>
                    
                    <div class="mt-6">
                        <div class="flex items-center justify-between mb-2">
                            <div class="flex items-center">
                                <div class="w-3 h-3 rounded-full bg-indigo-500 mr-2"></div>
                                <span class="text-sm text-gray-600 dark:text-gray-300">${state.userNames.user1}</span>
                            </div>
                            <div class="flex items-center">
                                <div class="w-3 h-3 rounded-full bg-green-500 mr-2"></div>
                                <span class="text-sm text-gray-600 dark:text-gray-300">${state.userNames.user2}</span>
                            </div>
                        </div>

                        ${(habit.user1 || habit.user2) ? `
                        <div class="flex flex-wrap gap-1">
                            ${Array.from({ length: Math.max(
                                habit.user1 ? habit.user1.chain.length : 0,
                                habit.user2 ? habit.user2.chain.length : 0
                            ) }, (_, i) => {
                                const u1 = habit.user1 && i < habit.user1.chain.length ? habit.user1.chain[i] : null;
                                const u2 = habit.user2 && i < habit.user2.chain.length ? habit.user2.chain[i] : null;
                                
                                return `
                                    <div class="relative w-3 h-6">
                                        ${u1 !== null ? `<div class="absolute bottom-0 w-3 ${u1 ? 'bg-indigo-500' : 'bg-indigo-200 dark:bg-indigo-800'} rounded-t-sm"

        function openEditHabitsModal() {
            editHabitsList.innerHTML = '';

            if (userHabits.length === 0) {

            } else {
                userHabits.forEach(habit => {
                    const streak = calculateStreak(habit.chain);
                    
                    const habitOption = document.createElement('div');
                    habitOption.className = 'edit-habits-selector bg-white dark:bg-gray-700 p-3 rounded-lg shadow cursor-pointer';
                    habitOption.dataset.habitId = habit.id;
                    
                    habitOption.innerHTML = `
                        <div class="flex justify-between items-center">
                            <h4 class="font-medium text-gray-800 dark:text-white">${habit.name}</h4>
                            <span class="text-sm font-medium ${streak > 3 ? 'text-green-500' : streak > 0 ? 'text-yellow-500' : 'text-gray-500'}">
                                Streak: ${streak} days
                            </span>
                        </div>
                        <div class="flex mt-2">
                            <span class="text-xs text-gray-600 dark:text-gray-300">Completed ${habit.chain.filter(d => d).length} of ${habit.chain.length} days</span>
